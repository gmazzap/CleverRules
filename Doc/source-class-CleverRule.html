  47:     function __construct( $type = '', $value = '' ) {

  53:     /**
  54:      * Do sanization. Return the sanitized value or null if sanitization fails
  55:      *
  56:      * @access protected
  57:      * @return mixed Sanitized value or false
  58:      * @uses Sanitize::sanitize_type
  59:      */
  60:     function sanitize() {
  61:         if ( is_null( $this->value ) ) return null;
  62:         if ( is_callable( $this->type ) ) return call_user_func( $this->type, $this->value );
  63:         if ( ! is_string( $this->type ) ) return null;
  64:         return $this->sanitize_type();
  65:     }

  68:     /**
  69:      * Handle sanitization based on type
  70:      *
  71:      * @return mixed Sanitized value or null
  72:      * @access protected
  73:      */
  74:     protected function sanitize_type() {
  75:         switch ( $this->type ) {
  76:             case 'int' :
  77:                 return is_numeric( $this->value ) && intval( $this->value ) ? $this->value : null;
  78:             case 'bool' :
  79:                 return (bool) $this->value;
  80:             case 'safe_string' :
  81:                 return preg_match( '/^[a-z0-9_\.]+$/', $this->value ) === 1 ? $this->value : null;
  82:             case 'callable' :
  83:                 return is_callable( $this->value ) ? $this->value : null;
  84:             case 'array' :
  85:                 $this->value = (array) wp_parse_args( $this->value );
  86:                 $f_values = array_filter( array_map( 'is_string', array_values( $this->value ) ) );
  87:                 if ( empty( $f_values ) ) return null;
  88:                 return $this->value;
  89:             case 'string_keyed_array' :
  90:                 $this->value = (array) wp_parse_args( $this->value );
  91:                 $f_keys = array_filter( array_map( 'is_string', array_keys( $this->value ) ) );
  92:                 $f_values = array_filter( array_map( 'is_string', array_values( $this->value ) ) );
  93:                 if ( empty( $f_keys ) || empty( $f_values ) ) return null;
  94:                 return $this->value;
  95:             case 'route' :
  96:                 if ( ! is_string( $this->value ) || ! substr_count( $this->value, '/' ) )
  97:                     return null;
  98:                 return trim( $this->value );
  99:             case 'file' :
 100:             case 'php_file' :
 101:                 $pi = pathinfo( $this->value );
 102:                 if ( empty( $pi ) || ! isset( $pi['filename'] ) ) return null;
 103:                 if ( $this->type == 'file') return $this->value;
 104:                 $ext = str_replace( '_file', '', $this->type );
 105:                 return isset( $pi['extension'] ) && ( $pi['extension'] == $ext )
 106:                     ? $this->value
 107:                     : null;
 108:             default :
 109:                 return null;
 110:         }
 111:     }

 523:      * Only function in the package that override a core one.
 524:      * Called on every frontend request and in some admin screens, this function create
 525:      * the query vars based on the url request.
 526:      *
 527:      * @param array $extra_query_vars   additional query vars WordPress pass to core parse_request
 528:      * @uses CleverRules::clever_unset_wp_rewrite
 529:      * @uses CleverRules::get_clever_rules

 531:      * @uses CleverRules::clever_pieces_match

 533:      * @uses CleverRules::to_wp
 534:      * @access public
 535:      */
 536:     function parse_request( $extra_query_vars = '' ) {
 537:         if ( ! defined( 'CLEVER_RULES' ) ) return;
 538:         $this->clever_unset_wp_rewrite();
 539:         self::$clever_vars = array_merge( $this->public_query_vars, $this->private_query_vars );
 540:         $the_rules = (array) self::get_clever_rules();
 541:         if ( empty( $the_rules ) ) return $this->to_wp( $extra_query_vars );
 542:         $full_url = add_query_arg( array() );
 543:         $sane_url = explode( '?', $full_url );
 544:         $_qs = array();
 545:         if ( isset( $sane_url[1] ) ) parse_str( $sane_url[1], $_qs );
 546:         $url = $sane_url[0];
 547:         $pieces = array_values( array_filter( explode( '/', $url ) ) );
 548:         $found = self::find_clever_rules( $the_rules, $pieces );
 549:         if ( $found[0] == 'home' )
 550:             return $this->clever_request( $found[1], array(), $_qs, $extra_query_vars );
 551:         if ( empty( $found[1] ) ) return $this->to_wp( $extra_query_vars );
 552:         $match_rule = self::clever_pieces_match( array_values( $found[1] ), $pieces );
 553:         if ( $match_rule === FALSE ) return $this->to_wp( $extra_query_vars );
 554:         return $this->clever_request( $match_rule[0], $match_rule[1], $_qs, $extra_query_vars );
 555:     }

 782:     /**

 837:     /**
 838:      * Static method used to check the url pieces against the rule pieces
 839:      *
 840:      * @param array $pieces the url pieces
 841:      * @param array $rule   the rule array
 842:      * @uses CleverRules::clever_piece
 843:      * @access private
 844:      * @return array    two items array, first is the number of matches,
 845:      *                  second is dynamic query vars found
 846:      */
 847:     private static function clever_pieces_find( $pieces, $rule ) {
 848:         $match = 0;
 849:         $query_i = array();
 850:         $stop = false;
 851:         $i = 0;
 852:         while ( ! empty( $pieces ) && ($stop == false) ) {
 853:             $piece = array_shift( $pieces );
 854:             if ( $rule['route'][$i] === $piece ) {
 855:                 $match ++;
 856:             } elseif ( substr_count( $rule['route'][$i], '%' ) == 1 ) {
 857:                 $q = self::clever_piece( $rule['route'][$i], $piece );
 858:                 if ( $q ) {
 859:                     $query_i[] = $q;
 860:                     $match ++;
 861:                 } else {
 862:                     $stop = true;
 863:                 }
 864:             } else {

 866:             }
 867:             $i ++;
 868:         }

 902:     private static function clever_piece_dyn( $route_piece, $url_piece, $type = '%d' ) {
 903:         $matches = array();
